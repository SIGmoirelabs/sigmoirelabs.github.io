[docs]    @override
    def read_documents(self, index_name: str):
        """
        Construct Elasticsearch documents from ``self.data``, which will be added to the index ``index_name``.

        This method overrides :meth:`elasticwrap.datasource.DataSource.read_documents`.

        Parameters
        ----------
        index_name : str
            The name of the Elasticsearch index.

        Yields
        ------
        Iterator[PostDocument[Dict[str, any]]]
            Iterator of documents that will be used by :meth:`documents.DocumentsWrapper.create_documents`.
        """
        for dic in self.data:
            doc: PostDocument = {
                "_index": index_name,
                "_source": dic
            }
            # add "_id" field if "uuid" is provided
            if "uuid" in dic:
                doc["_id"] = dic["uuid"]
            yield doc

    @override
    def convert_documents(data) -> Iterator[Dict]:
        """Summary

        Parameters
        ----------
        data : Iterable[Document]
            The documents to convert to a regular Dict

        Yields
        ------
        Iterator[Dict]
            The converted documents. This is essentially just the data found in _source
        """
        for doc in data:
            yield doc["_source"]
