'''
Holds the `SearchInterface` class. It primarly functions as a base
for more specialised classes to search the ES store.
'''

[docs]class SearchInterface(Generic[AnyDocument], NamespacedClient, metaclass=abc.ABCMeta):
    """
    A Client that is focused on searching. Should be implemented by more specific classes which
    know the structure of the data they are searching through and what the user is likely
    to want to search for.

    In general, a user can construct their search using methods found in this interface + the
    implementing classes. Once they are done constructing the search, they can retrieve the data.

    Implements a number of default methods which are useful for all search clients.

    Attributes
    ----------
    canonical_date_es_format : str
        The format to send to ES which uses a different formatting method to Python, set to yyyy-MM-dd
    canonical_date_format : str
        The format that dates should be presented in, set to %Y-%m-%d
    logger : Logger
        Every search client should have a logger. Ideally, one with a unique name is used,
        but if none is provided, SearchInterface comes with a default logger.
    namespace : str
        The namespace for a particular search client. This namespace defines on what
        indices the client will look by default (for example, "git" for the CommitClient).

        Indices are intended to follow the naming scheme {projectname}_{namespace}.
    query : Dict[str, Any]
        A dictionary that holds all the arguments that will be sent to elasticsearch. Usually,
        an end user should have no reason to see or interact with this dictionary.

    See Also
    --------
    CommitClient: Search Client focused on Git commits

    Notes
    -----
    The SearchInterface keeps track of what searches are being done with it and does not
    automatically clear out it's arguments. Meaning the subsequent searches will always
    yields the same results. If you want to reset the search, you can use the reset() method.

    SearchInterface is defined as generic for a specific document structure. Implementations
    are encouraged to define a TypedDict representation of the structure of the documents they
    deal with and to specify this representation on inheritence.
    """

    def __init__(self, client: Elasticsearch, name: str, log_name: str = "ElasticWrap_Search"):
        # The mypy annotation says Elasticsearch, but should actually be ElasticWrap. Did this to solve cyclical imports.
        """
        Constructor for the SearchInterface. Should be called with arguments by any client
        that implements it.

        Parameters
        ----------
        client : ElasticWrap
            The elasticwrap object to run searches with.
        name : str
            The namespace for this search client
        log_name : str, optional
            The name of the logger to use with this search client

        See Also
        --------
        elasticsearch._sync.client._base.NamespacedClient
        """
        super().__init__(client)
        self.namespace = "_" + name
        self.logger = logging.getLogger(log_name)
        self.query: Dict[str, Any] = self.reset()
        self.canonical_date_format = "%Y-%m-%d"
        self.canonical_date_es_format = "yyyy-MM-dd"

    def __indices_of_projects(self, proj: List[str]) -> str:
        if proj:
            return (self.namespace + ",").join(proj) + self.namespace
        else:
            self.logger.info("List of projects was empty, will search through all %s related indices.", self.namespace)
            return "*" + self.namespace


[docs]    def reset(self) -> Dict[str, Any]:
        """
        Resets the current search to simply getting all documents and sorting by whatever ES decides is correct.

        Returns
        -------
        Dict[str, Any]
            The query dict for inspection. Automatically sets self.query as well.
        """
        new_dict = {'query': RecursiveDict(), "sort": [{"_shard_doc": "asc"}]}
        self.query = new_dict
        return self.query


    def _search(self, indices: str, anonymize: bool, **kwargs) -> Iterator[List[AnyDocument]]:
        """
        Send off the search to ES and yield the results

        Parameters
        ----------
        indices : str
            The comma seperated string of indices to search on.
        anonymize : bool
            Whether or not to anonymize the resulting data.
        **kwargs
            Any extra arguments to pass along to the search.

        Returns
        -------
        Iterator[List[AnyDocument]]
            An iterator of documents as returned by ES.
        """
        merge_dicts('query', self.query, kwargs)
        merge_dicts('sort', self.query, kwargs)
        merge_dicts('_source', self.query, kwargs)
        self.logger.debug("Doing search with kwargs: %s", kwargs)
        if anonymize:
            self.logger.info("Starting data request process, will anonymize pages as they come in.")
            return self._anonymize_func(scan_pit, self._client, indices, **kwargs) # type: ignore
        else:
            self.logger.info("Starting data request process.")
            # We ignore the type here, since the resulting value of scan_pit is guaranteed to be AnyDocument
            # But there is no way to specify this in scan_pit cleanly.
            return scan_pit(self._client, indices, **kwargs) # type: ignore


[docs]    def search(self, proj: List[str] = [], anonymize: bool = False, **kwargs) -> Iterator[List[AnyDocument]]:
        """
        Start searching with the constructed query.

        Parameters
        ----------
        proj : list[str], optional
            The list of projects to search through
        anonymize : bool, optional
            Whether to anonymize the returned data, defaults to false
        **kwargs
            Additional keyword arguments to pass along to the vanilla elasticsearch search.

        Returns
        -------
        Iterator[List[AnyDocument]]
            A lazily paginated list of found documents. Will by default run in chunks of 1000 documents. This means
            that every time `next` is called, another, at most, 1000 documents are searched for and returned.

        Notes
        -----
        Just like with the vanilla elasticsearch search, a `query` dict can be passed as a keyword argument.
        This dict can have different search constraints from those constructed by the search client.
        Before the search, both sets of search constraints are merged together. If both the
        provided dict and the constructed dict try to add conflicting constraints, the constraints
        from the provided dict are used.

        This also means that the search construction can be circumvented entirely and the interface
        search can be directly used with a query object. This could be useful if you want to make
        use of the automatic pagination provided by this interface, but would prefer to construct
        the query yourself.

        It is possible that the query you pass along manually conflicts in a more fundamental way
        with the constructed query (for example, combining `match_all` and a `bool` filter causes issues). Therefore
        it is recommended to only combine the two sources if you know what you are doing.

        See Also
        --------
        elasticwrap.helpers.search_pit
        elasticsearch.search
        """
        index_string = self.__indices_of_projects(proj)
        return self._search(index_string, anonymize, **kwargs)



[docs]    def get_all(self, anonymize: bool = False, **kwargs) -> Iterator[List[AnyDocument]]:
        """
        Gets all documents for the associated namespace in the PAW, ignoring all the currently set filters.

        Parameters
        ----------
        anonymize : bool, optional
            Whether to anonymize the returned data, defaults to false.
        **kwargs
            Additional arguments to send to the search method

        Returns
        -------
        Iterator[List[AnyDocument]]
            A lazily paginated list of all the documents

        See Also
        --------
        search
        """
        old_query = self.query
        self.reset()
        resp = self.search([], anonymize, query={'match_all': {}}, **kwargs)
        self.query['query'] = old_query
        return resp



[docs]    def add_accepted_fields(self, fields: Set[Union[str, Dict]]) -> None:
        """
        Used to filter out what fields to return when searching.

        This does not filter out documents, it specifically filters
        out the fields in the returned documents. If you want to filter
        out documents, use `add_filters`.

        By default all fields are returned, as soon as fields are added
        using this method, only the accepted fields will be returned.

        Parameters
        ----------
        fields : set[str | Dict]
            The fields to return when searching.
            If the field you want to add is nested deeper, a dict must
            be provided.

        See Also
        --------
        add_filters
        """
        if '_source' in self.query:
            self.query['_source'] |= fields
        else:
            self.query['_source'] = fields



[docs]    def del_accepted_fields(self, fields: Set[Union[str, Dict]]) -> None:
        """
        Used to remove a set of fields from the set of accepted fields.

        This is only used to remove fields that were added to the accepted
        list using `add_accepted_fields`. Elasticsearch does not allow you to remove some
        fields from the results and keep the rest.

        Parameters
        ----------
        fields : set[str | Dict]
            The fields to no longer return when searching.

        See Also
        --------
        add_accepted_fields
        """
        if '_source' in self.query:
            self.query['_source'] -= fields
            if not self.query['_source']:
                del self.query['_source']
        else:
            self.logger.warning("Tried to remove accepted fields, but none were provided yet. Doing nothing")



[docs]    def add_filters(self, filters: List[Tuple[str, ElasticPrimitiveList]], negate=False) -> None:
        """
        Adds a filter to the search. Any document that is returned by the search must adhere
        to this filter.

        Parameters
        ----------
        filters : List[Tuple[str, ElasticPrimitiveList]]
            A list of filters to return. A filter takes the format
            (term, match) which means that the value of `term` must
            be found in the `match` list
        negate : bool
            If ``True``, will instead ensure that the values of ``term``
            are NOT found in the ``match`` list. Defaults to ``False``
        """
        for filt in filters:
            if negate:
                self.query['query']['bool']['must_not']['terms'][filt[0]] = filt[1]
            else:
                self.query['query']['bool']['filter']['terms'][filt[0]] = filt[1]



[docs]    def del_filters(self, filters: List[str]) -> None:
        """
        Remove all filters for a specific term.

        Parameters
        ----------
        filters : list[str]
            The terms to delete the filters from.

        See Also
        --------
        add_filters
        """
        for filt in filters:
            if filt in self.query['query']['bool']['filter']['terms']:
                del self.query['query']['bool']['filter']['terms'][filt]

        """
        Sets a script for filtering.

        Notes
        -----
        This should really only be used for comparing 2 fields in the same document,
        I.E. ``doc.val1 != doc.val2``. Every other case can be handled better in other ways.

        For more information see: For more information, see https://www.elastic.co/guide/en/elasticsearch/reference/current/query-dsl-script-query.html
        """
        self.query['query']['bool']['filter']['script']['script'] = script

[docs]    def add_range_filter(self, term: str, rang: Dict[str, str]) -> None:
        """
        Adds a filter to the search. Any document which does not fall into a
        certain range will be removed from the search.

        Parameters
        ----------
        term : str
            The term to compare with
        rang : Dict[str, str]
            A dictionary representing the range object, as defined in the ES docs[1]_

        [1] Elastic co. "Range Query" https://www.elastic.co/guide/en/elasticsearch/reference/current/query-dsl-range-query.html
        """
        self.query['query']['bool']['filter']['range'][term] = rang
        self.query['sort'].extend({term: 'asc'})

[docs]    def filter_date(self, term: str, dates: List[Tuple[LogicalComp, Union[str, datetime]]]):
        """
        Removes documents for which the value of some term is outside some date range.

        Parameters
        ----------
        term : str
            The term to filter on
        dates : list[tuple[LogicalComp, str | datetime]]
            A list of tuples where the first value is the comparison and the second value is the value to compare to.

        Raises
        ------
        ParserError: If `dateutil` was unable to parse the provided date string.
        OverflowError: If the provided date string would exceed C MAXINT (impressive).

        Notes
        -----
        If any of the provided dates in a tuple is a string, it will be attempted to be parsed using the `dateutil` python library.
        """
        filt = {"format": self.canonical_date_es_format}
        for comp, dat in dates:
            if isinstance(dat, str):
                dat = parse(dat, dayfirst=True)
            if not dat.tzname():
                dat = dat.astimezone()
            filt[comp.value] = dat.strftime(self.canonical_date_format)
            filt["time_zone"] = dat.tzname() # type: ignore
        self.add_range_filter(term, filt)



[docs]    @abc.abstractmethod
    def anonymize_data(self, data: SourceDocument) -> None:
        """
        Stub, should be specified per search client depending on the structure of the data.
        This method will make sure that the provided document (in the form of a dict)
        is stripped of all relevant PII. This happens in-place and is irreversible.

        Parameters
        ----------
        data : SourceDocument
            The document to anonymize

        Raises
        ------
        NotImplementedError
            Will raise this error by default if the search client has not implemented
            this method.
        """
        raise NotImplementedError


    def _anonymize_func(self, func: Callable[..., Iterator[List[AnyDocument]]], *args, **kwargs) -> Iterator[List[AnyDocument]]:
        for resp in func(*args, **kwargs):
            self.logger.info("Received page of data, anonymizing...")
            for point in resp:
                self.anonymize_data(point['_source'])
            yield resp

    @classmethod
    def __subclasshook__(cls, subclass):
        return (hasattr(subclass, 'anonymize_data') and
                callable(subclass.anonymize_data) or
                NotImplemented)

[docs]class UnstructuredSearchClient(GenericSearchClient):
    """
    A generic search client which does not think in terms of "projects" and "namespaces"

    Instead of providing a list of project names, you can directly provide a list of indices.

    Intended when you simply want to use ElasticWrap without any of the PAW assumptions.
    """

    @override
    def __indices_of_projects(self, proj: List[str]) -> str:
        return ",".join(proj)

    @override
    def __init__(self, *args, **kwargs):
        if "name" in kwargs:
            del kwargs["name"]
        super().__init__(*args, name="", **kwargs)
