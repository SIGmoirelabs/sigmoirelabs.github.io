[docs]class RecursiveDict(defaultdict):
    """
    This class creates a defaultdict which will automatically
    set the value of an unknown key to be a new recursive defaultdict
    when it is encountered for the first time.

    Additionally, if a key is deleted in the RecursiveDict which causes
    it to become empty, it will automatically delete the associated key
    from the parent (causing recursive deletions until either the root
    is encountered or a parent has another child).

    Example
    -------
    >>> a = RecursiveDict()
    >>> a['b']['c']['d']['e'] = 4
    >>> a['b']['f'] = 3
    >>> a
    RecursiveDict(..., {'b': RecursiveDict(..., {'f': 3, 'c': RecursiveDict(..., {'d': RecursiveDict(..., {'e': 4})})})})
    >>> del a['b']['c']['d']['e']
    >>> a
    RecursiveDict(..., {'b': RecursiveDict(..., {'f': 3})})

    See Also
    --------
    defaultdict

    Attributes
    ----------
    args :
        Arguments that were passed to this dict, will be passed to any automatically generated children.
    kwargs :
        Keyword arguments that were passed to this dict, will be passed to any automatically generated children.
    parent : Optional[Self]
        The parent of this dict. If set to None, that means this is the root of the tree.
    parent_key : TYPE
        The key with which this dict was stored in the parent.

    Notes
    -----
    On a copy operation, the tree slightly falls apart. If you have a layered RecursiveDict tree and you copy the root.
    The children will still be linked to eachother, only the first child is linked to the original parent and not the
    copy, resulting in the following behavior:
    >>> a[1][2][3] = 4
    >>> b = a.copy()
    >>> del a[1][2][3]
    >>> a -> {}
    >>> b -> {1: {}}

    If you make a deepcopy, the structure is kept intuitive and children are linked as expected.
    """

    def __init__(self, *args, parent: Optional[Self] = None, parent_key: Optional[Self] = None, **kwargs):
        self.parent = parent
        self.parent_key = parent_key
        self.args = args
        self.kwargs = kwargs
        super().__init__(*(args or (RecursiveDict,)), **kwargs)

    def __delitem__(self, key):
        super().__delitem__(key)
        if (len(self) == 0 and self.parent is not None and self.parent_key is not None):
            del self.parent[self.parent_key]

    def __missing__(self, key):
        val = RecursiveDict(*self.args, parent=self, parent_key=key, **self.kwargs)
        self[key] = val
        return val

    def __deepcopy__(self: Self, memo) -> Self:
        copy = RecursiveDict()
        for key in self:
            copy[key] = deepcopy(self[key], memo)
            if isinstance(copy[key], RecursiveDict):
                copy[key].parent = self
        return copy



[docs]def pop_transport_kwargs(kw: MutableMapping[str, Any]) -> Dict[str, Any]:
    """
    Extract kwargs commonly used to specify parameters for network operations.

    Parameters
    ----------
    kw : MutableMapping[str, Any]
        The kwargs to extract from

    Returns
    -------
    Dict[str, Any]
        The extracted kwargs
    """
    transport_kwargs = {}
    for key in ("headers", "api_key", "http_auth", "basic_auth", "bearer_auth"):
        try:
            value = kw.pop(key)
            if key == "http_auth":
                key = "basic_auth"
            transport_kwargs[key] = value
        except KeyError:
            pass
    return transport_kwargs



[docs]def normalize_from_keyword(kw: MutableMapping[str, Any]) -> None:
    """
    Setting query={"from": ...} would make 'from' be used as a keyword argument instead of 'from_'. We handle that here.
    """
    if "from" in kw:
        kw["from_"] = kw.pop("from")



[docs]def get_from_dict_dot(d: Mapping[str, Any], key: str) -> Any:
    """
    Method for getting from a dictionary using dot notation

    Example
    -------
    >>> key = "a.b.c"
    >>> d = {"a": {"b": {"c": 1}}}
    >>> get_from_dict_dot(d, key) == 1

    Parameters
    ----------
    d : Mapping[str, Any]
        The dictionary to get from
    key : str
        The dot notated string to get with

    Returns
    -------
    Any
        The data
    """
    keys = key.split(".")
    for k in keys:
        d = d[k]
    return d



[docs]def merge_dicts(key: str, source: Mapping[str, Any], dest: MutableMapping[str, Any]):
    """
    Merges two dicts together on a specific key. Happens in place

    Parameters
    ----------
    key : str
        The key to merge on.
    source : Mapping[str, Any]
        The "source" of the data.
    dest : MutableMapping[str, Any]
        The dictionary to put the data in.

    Notes
    -----
    If the key is not found in the source, nothing happens.

    If the key is found in the source but not the dest, a shallow copy
    is put directly in the dest

    If the key is found in both the source and the dest, the dictionaries
    are merged together. In case of a conflict, the dest will be assumed canonical.
    """
    if key in source:
        if key in dest:
            dest[key] = merge({}, source[key], dest[key])
        else:
            dest[key] = source[key].copy()
