def scan_pit(
    client: Elasticsearch,
    index: str,
    initial_keep_alive: str = "5m",
    extend_keep_alive: str = "1m",
    query: Dict[str, Any] = {},
    raise_on_error: bool = True,
    size: int = 1000,
    request_timeout: Optional[float] = None,
    clear_pit: bool = True,
    page_max = 0,
    **kwargs: Any,
) -> Iterator[List[Document]]:
    """
    Use the Point in time API to paginate search results.

    Arguments:
    ---------
    client : Elasticsearch
        the ES client object to connect to the cluster
    index : st
        The ES index to gather data from
    initial_keep_alive : str
        Specify how long a consistent view of the index should be maintained.
    extend_keep_alive : str
        how long the time to live of the point in time after each search should be extended
    query : Dict[str, Any]
        body for the ES api
    raise_on_error : bool
        raises an exception (``PitError``) if an error is encountered (some shards fail to execute).
    size : int
        size (per shard) of the batch send at each iteration.
    request_timeout : Optional[float]
        explicit timeout for each call to ``search_pit_helper``
    clear_pit : bool
        explicitly calls delete on the pit id via the clear
        pit API at the end of the method on completion or error, defaults
        to true.
    page_max : int
        How many pages you want to retrieve at maximum. If set to 0
        all pages are retrieved.

    See Also
    --------
    elasticsearch.helpers.scan

    Notes:
    ------
    PIT is the currently suggested way to paginate results in ElasticSearch[1]_.
    Because oftentimes we will be returning many documents, it has been decided
    to always default to this search. It is mostly intended for internal use, but
    can also be helpful when you want to search more manually.

    [1] Elastic co. "Paginate search results" https://www.elastic.co/guide/en/elasticsearch/reference/current/paginate-search-results.html
    """

    logger.debug("Setting up PIT")
    resp, search_kwargs, pit, new_query = __setup_pit(client, index, initial_keep_alive, extend_keep_alive, query, size, request_timeout, **kwargs)
    logger.debug("Done setting up PIT")
    q = CloseableQueue()
    logger.debug("Going to search with kwags: %s", search_kwargs)
    t = PropagatingThread(target=__do_search, args=(resp, pit, client, new_query, raise_on_error, clear_pit, page_max, q), kwargs=search_kwargs)
    logger.debug("Going to start thread")
    t.start()
    logger.debug("Getting results from queue")
    yield from dequeue(q)
    t.join() # Throw up any unhandled exceptions in the thread if they occured

def __setup_pit(
    client: Elasticsearch,
    index: str,
    initial_keep_alive: str,
    extend_keep_alive: str,
    query: Dict[str, Any],
    size: int,
    request_timeout: Optional[float],
    **kwargs: Any,
    ) -> Tuple[ObjectApiResponse[Any], Dict[str, Any], Dict[str, str], Optional[Dict[str, Any]]]:

    client = client.options(request_timeout=request_timeout, **pop_transport_kwargs(kwargs))
    client._client_meta = (("h", "s"),)

    normalize_from_keyword(kwargs)

    # Open point in time
    logger.debug("Going to open PIT")
    tmp = client.open_point_in_time(index=index, keep_alive=initial_keep_alive)
    logger.debug("PIT opened")
    pit = {
        "id": tmp["id"],
        "keep_alive": extend_keep_alive
    }
    try:
        search_kwargs = query.copy() # Should be copy since query is possibly mutable
        normalize_from_keyword(search_kwargs)
        search_kwargs.update(kwargs)
        search_kwargs["track_total_hits"] = False # this is set for speedup reasons
        search_kwargs["size"] = size
        logger.debug("Going to do first search")
        return client.search(pit=pit, **search_kwargs), search_kwargs, pit, None

    # Try the old deprecated way if we fail immediately on parameters.
    except TypeError:
        logger.warning("Failed on parameters, trying old deprecated way for searching")
        search_kwargs = kwargs.copy()
        search_kwargs["track_total_hits"] = False
        search_kwargs["size"] = size
        logger.debug("Going to do first search")
        return client.search(query=query, pit=pit, **search_kwargs), search_kwargs, pit, query

def __do_search(resp, pit: Dict, client: Elasticsearch, query: Optional[Dict[str, Any]], raise_on_error: bool, clear_pit: bool, page_max: int, out: CloseableQueue, **search_kwargs) -> None:
    logger.debug("Started search thread")
    try:
        count: int = 1
        while len(resp['hits']['hits']) == search_kwargs['size'] and (page_max == 0 or count < page_max): # We're gonna have to go again
            logger.debug("Getting a new page")
            # Default to 0 if the value isn't included in the response
            shards_info: Dict[str, int] = resp["_shards"]

            __check_shards(shards_info, raise_on_error, pit)

            out.put(resp["hits"]["hits"])

            count += 1

            search_after = resp["hits"]["hits"][-1]["sort"]

            resp = client.search(query=query, pit=pit, search_after=search_after, **search_kwargs)

            pit['id'] = resp['pit_id']
        __check_shards(resp["_shards"], raise_on_error, pit)
        out.put(resp['hits']['hits'])
    finally:
        logger.debug("Closing the queue")
        out.close()
        if pit["id"] and clear_pit:
            client.options(ignore_status=404).close_point_in_time(id=pit["id"])

def __check_shards(shards_info: Dict[str, int], raise_on_error: bool, pit):
    shards_successful = shards_info.get("successful", 0)
    shards_skipped = shards_info.get("skipped", 0)
    shards_total = shards_info.get("total", 0)

    # check if we have any errors
    if (shards_successful + shards_skipped) < shards_total:
        shards_message = "PIT search has only succeeded on %d (+%d skipped) shards out of %d."
        logger.warning(
            shards_message,
            shards_successful,
            shards_skipped,
            shards_total,
        )
        if raise_on_error:
            raise PitError(
                pit["id"],
                shards_message
                % (
                    shards_successful,
                    shards_skipped,
                    shards_total,
                ),
            )
