"""
Contains custom types and typing related methods.

from datetime import datetime
from enum import Enum
from typing import Any, Optional, TypeVar, Generic, Union, Mapping, List, Sequence, AbstractSet
from typing_extensions import TypedDict, NotRequired


[docs]class SourceDocument(TypedDict):
    """
    General class to model the _source document that is returned by search clients.

    If you want to define data **NOT** gathered by Perceval, you should extend this class.

    See Also
    --------
    Document
    """



[docs]class PercevalDataDocument(TypedDict):
    """
    A generic ``TypedDict`` intended to define actual data as gathered by Perceval.

    If you want to define data gathered by Perceval, you should extend this class.
    """


AnyPercevalDataDocument = TypeVar("AnyPercevalDataDocument", bound=PercevalDataDocument)


[docs]class PercevalSourceDocument(SourceDocument, Generic[AnyPercevalDataDocument]):
    """
    A model that pre-defines some metadata fields as created by Perceval.

    See Also
    --------
    SourceDocument
    PercevalDataDocument
    """
    backend_name: str
    backend_version: str
    perceval_version: str
    timestamp: float
    origin: str
    uuid: str
    updated_on: float
    classified_fields_filtered: Optional[List[str]]
    category: str
    search_fields: dict
    tag: str
    metadata__updated_on: str
    metadata__timestamp: str
    data: AnyPercevalDataDocument


AnySourceDocument = TypeVar("AnySourceDocument", bound=Union[SourceDocument, Mapping[str, Any]])


[docs]class PostDocument(TypedDict, Generic[AnySourceDocument]):
    """
    Model of fields that are required when posting to ES
    """
    _index: str
    _source: AnySourceDocument
    _id: NotRequired[str]



[docs]class Document(PostDocument[AnySourceDocument]):
    """
    General class to model es documents as returned by search clients.

    Specific documents should be defined by the relevant search client.
    """
    _type: str
    _score: float
    _id: str


AnyDocument = TypeVar("AnyDocument", bound=Document)

PercevalDocument = Document[PercevalSourceDocument]

UnstructuredDocument = Document[Mapping[str, Any]]
