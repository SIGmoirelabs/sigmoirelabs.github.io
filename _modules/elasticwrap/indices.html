"""
Object to deal with ES indices as they are used in the PAW.
"""

import logging
from typing import Optional, List

from elasticsearch8._sync.client.indices import IndicesClient


[docs]class IndicesWrapper(IndicesClient):
    """
    Object to deal with ES indices as they are used in the PAW.

    See Also
    --------
    elasticsearch.client.IndicesClient

    Notes
    -----
    Has some underlying assumptions about how indices are named and what their purpose is.
    If indices are called differently, you can still use this object by using the vanilla ES methods, but most methods are not
    made with it in mind.

    It is assumed that all SIGmoirelab created indices will follow the structure `{proj_name}_{source}`. I.E. `FooCo_git`.

    Attributes
    ----------
    logger :
        A specific logger for this object
    """

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.logger = logging.getLogger("Elasticwrap_Indices")

    def _search_indices(self, query: str) -> List[str]:
        """
        Find all indices matching a certain query string.

        Parameters
        ----------
        query : str
            The query to search with.

        Returns
        -------
        list[str]
            A list of all indices that match the query.
        """
        return list(self.get_alias(index=query).keys())

    def _collapse_indices(self, indices: List[str]) -> str:
        """
        Collapses a list of indices into a comma seperated string (as expected by ES)

        Parameters
        ----------
        indices : list[str]
            The list of indices to collapse

        Returns
        -------
        str
            The string as expected by ES
        """
        return ",".join(indices)


[docs]    def search_indices(self, query: str) -> str:
        """
        Find all indices matching a certain query string.

        Parameters
        ----------
        query : str
            The query to search with.

        Returns
        -------
        str
            All the indices that match the query. Comma separated as ES expects.

        See Also
        --------
        _search_indices: Returns indices in a list instead of a comma separated string
        """
        return self._collapse_indices(self._search_indices(query))

        return self._search_indices("*")



[docs]    def get_current_kibana(self) -> Optional[str]:
        """
        Tries to determine what the current kibana index is.

        Returns
        -------
        Optional[str]
            The current kibana index, or None if there is none.

        Notes
        -----
        If there are multiple indices, a warning will be logged and the first will be returned.
        """
        resp = self.search_indices(".kibana")
        if len(resp) == 0:
            return None
        elif len(resp) > 1:
            self.logger.warning("Multiple indices aliased to Kibana, this should be impossible. Will return the first one found")
        return resp[0]

        return self.search_indices(proj + "_*")



[docs]    def get_indices_of_source(self, source: str) -> str:
        """
        Returns all the indices that belong to a specific data source.

        Parameters
        ----------
        source : str
            The data source to which the indices should belong.

        Returns
        -------
        str
            A comma separated list of indices that belong to this data source.
        """
        return self.search_indices("*_" + source)



[docs]    def create_index(self, proj: str, source: str):
        """
        Creates an index for a certain project and source.

        Parameters
        ----------
        proj : str
            The project name
        source : str
            The source name
        """
        resp = self.get_indices_of_project(proj)
        if len(resp) == 0:
            self.logger.warning('Project "%s" does not exist, but the new index "%s_%s" will still be created.', proj, proj, source)
        self.create(index=f'{proj}_{source}')

        """

        """
        self.delete(index=proj + '_*')
